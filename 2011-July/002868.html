I did remaster kiwi images for faster testing, here are my notes:

# setup
cd /tmp
mkdir -p remaster/{mnt,livecd}
cd remaster
mount -o loop PRESENSE_EWS-100.x86_64-1.5.0.iso mnt
cp -a mnt/* livecd
umount mnt
unclicfs livecd/PRESENSE_EWS-100-read-only.x86_64-1.5.0

# grow the image a bit
dd if=/dev/zero bs=1M count=100 >> fsdata.ext3
e2fsck -f fsdata.ext3
resize2fs fsdata.ext3
# that's the setup part

# now you can mount it, modify it and write it back to the card
mount -o loop fsdata.ext3 mnt

# modify the image
rpm --root `pwd`/mnt --nodeps -U /tmp/psm_app_cs_sen-1.5.2pre-1.x86_64.rpm

# remaster the clicfs
umount mnt
mkclicfs fsdata.ext3 livecd/PRESENSE_EWS-100-read-only.x86_64-1.5.0

# remaster the iso
genisoimage -o livecd.iso -b boot/x86_64/loader/isolinux.bin -c boot/x86_64/loader/boot.cat -no-emul-boot -boot-load-size 4 -boot-info-table -R livecd
isohybrid --id "$(cat livecd/boot/grub/mbrid)" --type 0x83 livecd.iso # make it bootable as a hdd

# test it :)
qemu -cdrom livecd.iso

Hope that helps!
